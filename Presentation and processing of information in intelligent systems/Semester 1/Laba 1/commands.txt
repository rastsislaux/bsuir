учить:
"создать папку cmd" - mkdir
"удалить папку cmd" - del
"создать папку linux" - mkdir
"удалить папку linux" - rm -rf
"создать файл cmd" - touch
"удалить файл cmd" - del
"создать файл linux" - touch
"удалить файл linux" - rm
"переместить папку cmd" - move
"переместить папку linux" - mv
"скопировать файл cmd " - copy
"скопировать файл linux" - cp
"переместить файл cmd" - move
"переместить файл linux" - mv
"вывод списка файлов в указанном каталоге cmd" - dir
"вывод списка файлов в указанном каталоге linux" - ls
"графически отобразить структуру каталогов диска или пути cmd" - tree
"создать каталог" - mkdir (win) / mkdir -p (linux)
"изменить права доступа к файлу linux" - chmod 
"отобразить текущий путь linux" - pwd
